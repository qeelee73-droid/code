CREATE DATABASE dbmslab;
USE dbmslab;

CREATE TABLE Customer(
C_id INT PRIMARY KEY,
cname VARCHAR(30),
city VARCHAR(30)
);

CREATE TABLE Account(
Acc_no INT PRIMARY KEY,
C_id INT,
Acc_Type VARCHAR(20),
Amount INT,
FOREIGN KEY(C_id) REFERENCES Customer(C_id)
);
