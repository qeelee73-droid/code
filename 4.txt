CREATE TABLE Stud (
    Roll NUMBER PRIMARY KEY,
    Att NUMBER,
    Status VARCHAR2(2)
);

DECLARE
    v_roll Stud.Roll%TYPE;
    v_att  Stud.Att%TYPE;

BEGIN

    v_roll := &roll_no;


    SELECT Att INTO v_att
    FROM Stud
    WHERE Roll = v_roll;


    IF v_att < 75 THEN
        UPDATE Stud
        SET Status = 'D'
        WHERE Roll = v_roll;

        DBMS_OUTPUT.PUT_LINE('Term not granted');

    ELSE
        UPDATE Stud
        SET Status = 'ND'
        WHERE Roll = v_roll;

        DBMS_OUTPUT.PUT_LINE('Term granted');
    END IF;

    COMMIT;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Student not found');

    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;
