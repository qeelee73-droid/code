CREATE TABLE Library (
    book_id NUMBER PRIMARY KEY,
    book_name VARCHAR2(50),
    author VARCHAR2(50),
    price NUMBER
);

CREATE TABLE Library_Audit (
    book_id NUMBER,
    book_name VARCHAR2(50),
    author VARCHAR2(50),
    price NUMBER,
    action_type VARCHAR2(10), 
    action_time DATE
);

CREATE OR REPLACE TRIGGER trg_library_audit
BEFORE UPDATE OR DELETE ON Library
FOR EACH ROW
BEGIN
    IF UPDATING THEN
        INSERT INTO Library_Audit
        VALUES (
            :OLD.Book_id,
            :OLD.Book_Name,
            :OLD.Author,
            :OLD.Price,
            'UPDATE',
            SYSDATE
        );

    ELSIF DELETING THEN
        INSERT INTO Library_Audit
        VALUES (
            :OLD.Book_id,
            :OLD.Book_Name,
            :OLD.Author,
            :OLD.Price,
            'DELETE',
            SYSDATE
        );
    END IF;
END;
