<!DOCTYPE html>
<html>
<head>
    <title>DOM Example</title>

    <script>
        // Function to change content dynamically
        function changeContent() {

            // Access element using its ID
            var element = document.getElementById("demo");

            // Change the text inside the element
            element.innerHTML = "Hello! This content is changed using JavaScript DOM.";

            // Change style dynamically
            element.style.color = "blue";
            element.style.fontSize = "20px";
        }
    </script>
</head>
<body>

    <h2>JavaScript DOM Example</h2>

    <!-- This paragraph will be changed dynamically -->
    <p id="demo">Original Content</p>

    <!-- Button to trigger JavaScript function -->
    <button onclick="changeContent()">Click Me</button>

</body>
</html>
